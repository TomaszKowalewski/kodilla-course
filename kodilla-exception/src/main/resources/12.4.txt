SELECT I.NAME, I2.SUMMARY, I2.DESCRIPTION
FROM ISSUESLISTS I
JOIN ISSUES I2 on I.ID = I2.ISSUESLISTS_ID;

SELECT U.FIRSTNAME, U.LASTNAME, I.SUMMARY, I.DESCRIPTION
FROM ISSUES I
JOIN USERS U on U.ID = I.USER_ID_ASSIGNEDTO;

SELECT U.ID, U.FIRSTNAME, U.LASTNAME, COUNT(I.ID) AS TASKS_NUMBER
FROM USERS U, ISSUES I
WHERE U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY U.ID
HAVING COUNT(I.ID) > 0;
